%%

%class ParserHelper;
%namespace frontend ;

%nonassoc LT;

%left PLUS;
%left TIMES;

%right POW;

%token ID;

%root expr;

%%

expr : expr PLUS expr  ~> {add}
     | expr TIMES expr ~> {times}
     | expr POW expr   ~> {power}
     | expr LT expr    ~> {compare}
     | ID              ~> {id}
;
