%%

%class LexerHelper ;

%namespace frontend ;

%%

"=" ~> eq ;
";" ~> semi ;
":" ~> colon ;
"," ~> comma ;
"\(" ~> lparen ;
"\)" ~> rparen ;
"module" ~> module ;
"procedure" ~> procedure;
"function" ~> function;
"begin" ~> begin ;
"end" ~> end ;
"type" ~> type ;
"var" ~> var ;
"record" ~> record ;
"if" ~> iff ;
"else" ~> elsee;
"elseif" ~> elseif;
"while" ~> whilee;
"do" ~> doo;
"then" ~> then;
"[A-Za-z_][A-Za-z0-9_]*" ~> id ; 
"[\n\t ]+" ;
